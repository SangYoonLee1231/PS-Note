```python
m, n = tuple(map(int, input().split()))

def isPrime(a):
    if a == 1 : return False
    i = 2
    while i*i <= a:
        if a % i == 0:
            return False
        i += 1
    return True

for x in range(m, n+1):
    if(isPrime(x)):
        print(x)
```
